阅读一个后缀表达式的方法是：从左往右读式子，一旦遇到运算符，就往前取 $n$ 个数（这个 $n$ 取决于运算符有多少个参数），然后擦掉这些参数和这个运算符，把计算结果写在那里。接下来，重复刚才的操作，直到表达式中只剩下一个数为止。

```cpp
#include <bits/stdc++.h>
using namespace std;

stack<int> n;
int s=0,x,y;

int main()
{
    char ch;
    do
    {
        ch=getchar();
        if(ch>='0'&&ch<='9') s=s*10+ch-'0';
        else if(ch=='.') n.push(s),s=0;
        else if(ch!='@')
        {
            x=n.top();
            n.pop();
            y=n.top();
            n.pop();
            switch (ch)
            {
                case'+':n.push(x+y);break;
                case'-':n.push(y-x);break;
                case'*':n.push(x*y);break;
                case'/':n.push(y/x);break;
            }
        }
    }while(ch!='@');
    
    printf("%d\n",n.top());

    return 0;   
}
```
